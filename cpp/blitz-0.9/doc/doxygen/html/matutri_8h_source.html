41 class UpperTriangularIterator {
42 public:
43  UpperTriangularIterator(unsigned rows, unsigned cols)
44  {
45  BZPRECONDITION(rows == cols);
46  size_ = rows;
47  good_ = true;
48  offset_ = 0;
49  i_ = 0;
50  j_ = 0;
51  }

56  {
